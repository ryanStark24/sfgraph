```python
# src/sfgraph/storage/base.py (defined in Plan 03)
from abc import ABC, abstractmethod
from typing import Any

class GraphStore(ABC):
    @abstractmethod
    async def merge_node(self, label: str, key_props: dict[str, Any], all_props: dict[str, Any]) -> str: ...
    @abstractmethod
    async def merge_edge(self, src_qualified_name: str, src_label: str, rel_type: str, dst_qualified_name: str, dst_label: str, props: dict[str, Any]) -> None: ...
    @abstractmethod
    async def query(self, cypher: str, params: dict[str, Any] | None = None) -> list[dict[str, Any]]: ...
    @abstractmethod
    async def get_labels(self) -> list[str]: ...
    @abstractmethod
    async def get_relationship_types(self) -> list[str]: ...
    @abstractmethod
    async def close(self) -> None: ...
```

    ```python
    # tests/test_falkordb_store.py
    """Integration tests for FalkorDBStore: MERGE round-trips and concurrent write safety."""
    import asyncio
    import pytest
    from sfgraph.storage.falkordb_store import FalkorDBStore
    from sfgraph.storage.base import GraphStore


    @pytest.fixture
    async def store(tmp_graph_db_path: str) -> FalkorDBStore:
        s = FalkorDBStore(tmp_graph_db_path)
        await s.start()
        yield s
        await s.close()


    def test_falkordb_store_is_graphstore(tmp_graph_db_path: str):
        s = FalkorDBStore(tmp_graph_db_path)
        assert isinstance(s, GraphStore)


    async def test_merge_node_and_query_round_trip(store: FalkorDBStore):
        qn = await store.merge_node(
            "ApexClass",
            {"qualifiedName": "TestClass"},
            {"qualifiedName": "TestClass", "name": "TestClass", "isTest": False},
        )
        assert qn == "TestClass"
        results = await store.query(
            "MATCH (n:ApexClass {qualifiedName: $qn}) RETURN n.name AS name",
            {"qn": "TestClass"},
        )
        assert len(results) == 1
        assert results[0]["name"] == "TestClass"


    async def test_merge_node_is_idempotent(store: FalkorDBStore):
        for _ in range(3):
            await store.merge_node(
                "ApexClass",
                {"qualifiedName": "Idem"},
                {"qualifiedName": "Idem", "name": "Idem"},
            )
        results = await store.query(
            "MATCH (n:ApexClass {qualifiedName: 'Idem'}) RETURN count(n) AS cnt"
        )
        assert results[0]["cnt"] == 1


    async def test_merge_edge_creates_relationship(store: FalkorDBStore):
        await store.merge_node("ApexClass", {"qualifiedName": "Caller"}, {"qualifiedName": "Caller", "name": "Caller"})
        await store.merge_node("ApexClass", {"qualifiedName": "Callee"}, {"qualifiedName": "Callee", "name": "Callee"})
        await store.merge_edge(
            "Caller", "ApexClass", "CALLS", "Callee", "ApexClass",
            {"confidence": 0.9, "resolutionMethod": "static", "edgeCategory": "CONTROL_FLOW", "contextSnippet": "Callee.run()"},
        )
        results = await store.query(
            "MATCH (a:ApexClass {qualifiedName: 'Caller'})-[r:CALLS]->(b:ApexClass {qualifiedName: 'Callee'}) RETURN r.confidence AS conf"
        )
        assert len(results) == 1
        assert abs(results[0]["conf"] - 0.9) < 0.001


    async def test_get_labels_returns_list(store: FalkorDBStore):
        await store.merge_node("ApexClass", {"qualifiedName": "LabelTest"}, {"qualifiedName": "LabelTest"})
        labels = await store.get_labels()
        assert isinstance(labels, list)
        assert "ApexClass" in labels


    async def test_concurrent_writes_do_not_corrupt(store: FalkorDBStore):
        count = 20
        tasks = [
            store.merge_node(
                "ApexClass",
                {"qualifiedName": f"Concurrent{i}"},
                {"qualifiedName": f"Concurrent{i}", "name": f"Concurrent{i}"},
            )
            for i in range(count)
        ]
        await asyncio.gather(*tasks)
        results = await store.query(
            "MATCH (n:ApexClass) WHERE n.qualifiedName STARTS WITH 'Concurrent' RETURN count(n) AS cnt"
        )
        assert results[0]["cnt"] == count
    ```

    IMPORTANT — FalkorDB Cypher for merge_node:
    FalkorDB uses standard Cypher MERGE. Build the MERGE statement using parameterized queries.
    The key_props become the MERGE match pattern; all_props become SET properties.
    Return n.qualifiedName from the MERGE so the method can return it.

    Example pattern:
    ```python
    async def merge_node(self, label: str, key_props: dict[str, Any], all_props: dict[str, Any]) -> str:
        key_match = ", ".join(f"{k}: ${k}" for k in key_props)
        set_clause = ", ".join(f"n.{k} = ${k}" for k in all_props)
        cypher = f"MERGE (n:{label} {{{key_match}}}) SET {set_clause} RETURN n.qualifiedName AS qn"
        params = dict(key_props) | dict(all_props)
        result = await self._write(cypher, params)
        # result is FalkorDB result object — extract first row, first column
        if result and result.result_set:
            return result.result_set[0][0] or ""
        return all_props.get("qualifiedName", "")
    ```

    ```python
    # src/sfgraph/storage/vector_store.py
    # Source: qdrant-client 1.17.1 docs + fastembed 0.8.0 docs
    import logging
    from qdrant_client import QdrantClient
    from qdrant_client.models import Distance, VectorParams, PointStruct

    logger = logging.getLogger(__name__)

    COLLECTION_NAME = "source_chunks"
    VECTOR_DIM = 384  # BAAI/bge-small-en-v1.5 output dimension


    class VectorStore:
        def __init__(self, path: str | None = None, url: str | None = None) -> None:
            """
            path=":memory:" → in-memory Qdrant (tests only, no persistence)
            path="/abs/path" → local file-backed Qdrant (dev / small orgs)
            url="http://localhost:6333" → Qdrant server/subprocess (large orgs)
            """
            if url:
                self._client = QdrantClient(url=url)
            elif path:
                self._client = QdrantClient(path=path)
            else:
                raise ValueError("Either path or url must be provided to VectorStore")
            # Import fastembed lazily to defer model download until first use
            self._embedder = None

        def _get_embedder(self):
            if self._embedder is None:
                from fastembed import TextEmbedding
                logger.info("Loading BAAI/bge-small-en-v1.5 embedding model (first use — may download ~130MB)")
                self._embedder = TextEmbedding("BAAI/bge-small-en-v1.5")
            return self._embedder

        async def initialize(self) -> None:
            """Create the source_chunks collection if it does not already exist."""
            existing = [c.name for c in self._client.get_collections().collections]
            if COLLECTION_NAME not in existing:
                self._client.create_collection(
                    collection_name=COLLECTION_NAME,
                    vectors_config=VectorParams(size=VECTOR_DIM, distance=Distance.COSINE),
                )
                logger.info("Created Qdrant collection: %s", COLLECTION_NAME)
            else:
                logger.debug("Qdrant collection already exists: %s", COLLECTION_NAME)

        async def upsert(self, node_id: str, text: str, payload: dict) -> None:
            """Embed text and upsert a point for node_id."""
            embedder = self._get_embedder()
            vectors = list(embedder.embed([text]))
            self._client.upsert(
                collection_name=COLLECTION_NAME,
                points=[
                    PointStruct(
                        id=abs(hash(node_id)) % (2**63),
                        vector=vectors[0].tolist(),
                        payload=payload | {"node_id": node_id},
                    )
                ],
            )

        async def search(self, query_text: str, limit: int = 10) -> list[dict]:
            """Find the top-k most similar source code chunks to query_text."""
            embedder = self._get_embedder()
            vectors = list(embedder.embed([query_text]))
            results = self._client.search(
                collection_name=COLLECTION_NAME,
                query_vector=vectors[0].tolist(),
                limit=limit,
            )
            return [
                {
                    "node_id": r.payload["node_id"],
                    "score": r.score,
                    "payload": r.payload,
                }
                for r in results
            ]
    ```
